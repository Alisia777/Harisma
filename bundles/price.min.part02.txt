  function allDatesForMarket(market) {
    const seen = new Set();
    const dates = [];
    baseRowsForMarket(market).forEach((row) => {
      (row.monthly || []).forEach((item) => {
        const value = isoDate(item.date);
        if (value && !seen.has(value)) {
          seen.add(value);
          dates.push(value);
        }
      });
    });
    dates.sort();
    return dates;
  }

  function presetRange(preset, market) {
    const dates = allDatesForMarket(market);
    if (!dates.length) return { dateFrom: "", dateTo: "" };
    const min = dates[0];
    const max = dates[dates.length - 1];
    if (preset === "all") return { dateFrom: min, dateTo: max };
    const maxDate = new Date(max);
    if (preset === "prev_month") {
      const start = new Date(Date.UTC(maxDate.getUTCFullYear(), maxDate.getUTCMonth() - 1, 1));
      const end = new Date(Date.UTC(maxDate.getUTCFullYear(), maxDate.getUTCMonth(), 0));
      return { dateFrom: start.toISOString().slice(0, 10), dateTo: end.toISOString().slice(0, 10) };
    }
    if (preset === "month") {
      const start = new Date(Date.UTC(maxDate.getUTCFullYear(), maxDate.getUTCMonth(), 1));
      return { dateFrom: start.toISOString().slice(0, 10), dateTo: max };
    }
    const days = PRESETS[preset]?.days || 30;
    const start = new Date(maxDate);
    start.setUTCDate(start.getUTCDate() - (days - 1));
    const startIso = start.toISOString().slice(0, 10);
    return { dateFrom: startIso < min ? min : startIso, dateTo: max };
  }

  function withinRange(value) {

    const to = state.priceWorkbench.filters.dateTo || "";

  function computePlanMetrics(row, series) {
    const rangeFrom = state.priceWorkbench.filters.dateFrom || "";
    const rangeTo = state.priceWorkbench.filters.dateTo || "";
    if (!rangeFrom || !rangeTo || rangeFrom > rangeTo) {
      return {
        planUnitsSelected: null,
        actualUnitsSelected: null,
        completionPct: null,
        selectedDays: 0,
        actualRevenueSelected: null,
        actualSource: "",
        planSource: "",
      };
    }
    const rangeStart = dateKeyToUtc(rangeFrom);
    const rangeEnd = dateKeyToUtc(rangeTo);
    if (!rangeStart || !rangeEnd) {
      return {
        planUnitsSelected: null,
        actualUnitsSelected: null,
        completionPct: null,
        selectedDays: 0,
        actualRevenueSelected: null,
        actualSource: "",
        planSource: "",
      };
    }
    const selectedDays = Math.floor((rangeEnd - rangeStart) / 86400000) + 1;
    const support = row.supportRow || null;
    const planFromSupport = sumMonthlyBuckets(support?.planMonths, rangeStart, rangeEnd);
    const fallbackPlanUnits = parseNum(row.planMonthUnits);
    const fallbackPlanDays = parseNum(row.planMonthDays) || 30;
    const planUnitsSelected = planFromSupport?.units != null
      ? planFromSupport.units
      : (fallbackPlanUnits && fallbackPlanDays ? (fallbackPlanUnits / fallbackPlanDays) * selectedDays : null);
    const actualPool = series.filter((item) => {
      const itemDate = isoDate(item.date);
      return itemDate && itemDate >= rangeFrom && itemDate <= rangeTo;
    });
    const actualValues = actualPool
      .map((item) => parseNum(item.ordersUnits) ?? parseNum(item.deliveredUnits))
      .filter((value) => value != null);
    const actualRevenueValues = actualPool
      .map((item) => parseNum(item.revenue))
      .filter((value) => value != null);
    let actualUnitsSelected = actualValues.length ? actualValues.reduce((sum, value) => sum + value, 0) : null;
    let actualRevenueSelected = actualRevenueValues.length ? actualRevenueValues.reduce((sum, value) => sum + value, 0) : null;
    let actualSource = actualUnitsSelected != null ? "daily" : "";
    if (actualUnitsSelected == null) {
      const monthlyActual = sumMonthlyBuckets(support?.actualMonths, rangeStart, rangeEnd);
      if (monthlyActual?.units != null) {
        actualUnitsSelected = monthlyActual.units;
        actualRevenueSelected = monthlyActual.revenue;
        actualSource = "monthly";
      }
    }
    const completionPct = actualUnitsSelected != null && planUnitsSelected > 0 ? actualUnitsSelected / planUnitsSelected : null;
    return {
      planUnitsSelected,
      actualUnitsSelected,
      completionPct,
      selectedDays,
      actualRevenueSelected,
      actualSource,
      planSource: planFromSupport?.units != null ? "monthly_plan" : (planUnitsSelected != null ? "seed" : ""),
    };
  }

  function applyEntry(row, entry) {
    const merged = entry ? normalizeEntry(entry) : normalizeEntry({ marketplace: row.marketplace, articleKey: row.articleKey, owner: row.owner });
    const series = (Array.isArray(row.monthly) ? row.monthly : [])
      .filter((item) => withinRange(item.date))
      .sort((left, right) => String(left?.date || "").localeCompare(String(right?.date || "")));
    const allowedMargin = resolveAllowedMargin(row);
    const effectiveAllowed = merged.allowedMarginManualPct != null ? merged.allowedMarginManualPct : allowedMargin.value;
    const latest = series.length ? series[series.length - 1] : null;
    const avgTurnover = series.length ? meanOrNull(series.map((item) => parseNum(item.turnoverDays))) : parseNum(row.turnoverCurrentDays);
    const latestPrice = latest ? parseNum(latest.price) : parseNum(row.currentFillPrice);
    const latestClient = latest ? (parseNum(latest.clientPrice) ?? (latestPrice != null && parseNum(latest.sppPct) != null ? latestPrice * (1 - parseNum(latest.sppPct)) : null)) : parseNum(row.currentClientPrice);
    const latestSpp = latest ? parseNum(latest.sppPct) : parseNum(row.currentSppPct);
    const historyMeta = buildHistoryMeta(series, row);
    const plan = computePlanMetrics(row, series);
    const periodPriceStart = series.length ? parseNum(series[0].price) : latestPrice;
    const periodClientStart = series.length ? (parseNum(series[0].clientPrice) ?? latestClient) : latestClient;
    const periodAvgPrice = series.length ? meanOrNull(series.map((item) => parseNum(item.price))) : latestPrice;
    const periodAvgClientPrice = series.length ? meanOrNull(series.map((item) => parseNum(item.clientPrice) ?? null)) : latestClient;
    const periodPriceDeltaPct = periodPriceStart != null && latestPrice != null && periodPriceStart > 0
      ? (latestPrice - periodPriceStart) / periodPriceStart
      : null;
    const periodClientDeltaPct = periodClientStart != null && latestClient != null && periodClientStart > 0
      ? (latestClient - periodClientStart) / periodClientStart
      : null;
    return {
      ...row,
      decisionId: merged.id,
      decisionStatus: merged.decisionStatus,
      targetFillPrice: merged.targetFillPrice ?? row.seedTargetFillPrice ?? latestPrice,
      targetClientPrice: merged.targetClientPrice ?? row.seedTargetClientPrice ?? latestClient,
      allowedMarginManualPct: merged.allowedMarginManualPct,
      effectiveAllowedMarginPct: effectiveAllowed,
      allowedMarginBaselinePct: allowedMargin.value,
      allowedMarginSourceBaseline: allowedMargin.source,
      allowedMarginNote: allowedMargin.note,
      allowedMarginSource: sourceLabel({ ...row, allowedMarginManualPct: merged.allowedMarginManualPct }),
      currentFillPriceDisplay: latestPrice,
      currentClientPriceDisplay: latestClient,
      currentSppDisplay: latestSpp,
      valueDate: isoDate(latest?.date) || historyMeta.freshnessDate || "",
      clientDiscountPct: latestPrice != null && latestClient != null && latestPrice > 0 ? 1 - (latestClient / latestPrice) : null,
      periodPriceStartDisplay: periodPriceStart,
      periodClientPriceStartDisplay: periodClientStart,
      periodAvgFillPriceDisplay: periodAvgPrice,
      periodAvgClientPriceDisplay: periodAvgClientPrice,
      periodPriceDeltaPct,
      periodClientDeltaPct,
      turnoverDisplay: avgTurnover,
      monthlyFiltered: series,
      historyMode: historyMeta.mode,
      historyFreshnessDisplay: historyMeta.freshnessDate,
      historyNoteDisplay: historyMeta.note,
      historyHasDailyFact: historyMeta.mode === "daily",
      owner: merged.owner || row.owner || "",
      dueDate: merged.dueDate || "",
      needsTask: merged.needsTask,
      escalation: merged.escalation,
      reason: merged.reason || row.seedReason || "",
      comment: merged.comment || "",
      updatedBy: merged.updatedBy,
      updatedAt: merged.updatedAt,
      createdAt: merged.createdAt,
      planUnitsSelected: plan.planUnitsSelected,
      actualUnitsSelected: plan.actualUnitsSelected,
      completionPctSelected: plan.completionPct,
      selectedDays: plan.selectedDays,
      periodRevenueSelected: plan.actualRevenueSelected,

    };
  }

  function signalsForRow(row) {
    const items = [];
    if (row.marginTotalPct != null && row.effectiveAllowedMarginPct != null && Number(row.marginTotalPct) < Number(row.effectiveAllowedMarginPct)) items.push({ label: "Маржа ниже допустимой", kind: "danger" });
    if (row.turnoverDisplay != null && Number(row.turnoverDisplay) >= 60) items.push({ label: "Высокая оборачиваемость", kind: "danger" });
    else if (row.turnoverDisplay != null && Number(row.turnoverDisplay) >= 40) items.push({ label: "Оборачиваемость требует внимания", kind: "warn" });
    if (row.completionPctSelected != null && Number(row.completionPctSelected) < 0.8) items.push({ label: "Отстаёт от плана", kind: "warn" });
    if (row.targetFillPrice != null && row.hardMinPrice != null && Number(row.targetFillPrice) < Number(row.hardMinPrice)) items.push({ label: "Новая цена ниже жёсткого минимума", kind: "danger" });
    if (row.needsTask) items.push({ label: "Есть задача", kind: "info" });
    if (row.historyNote) items.push({ label: "Проверь свежесть факта", kind: "info" });
    return items;
  }

  function riskScore(row) {
    return signalsForRow(row).reduce((sum, item) => sum + (item.kind === "danger" ? 3 : item.kind === "warn" ? 2 : 1), 0);
  }

  function filteredRows() {
    const search = String(state.priceWorkbench.filters.search || "").trim().toLowerCase();
    const ownerFilter = state.priceWorkbench.filters.owner || "all";
    const statusFilter = state.priceWorkbench.filters.status || "all";
    const sortMode = state.priceWorkbench.filters.sort || "risk";
    const riskOnly = Boolean(state.priceWorkbench.filters.riskOnly);
    const rows = effectiveRows().filter((row) => {
      if (search) {
        const haystack = [
          row.articleKey,
          row.name,
          row.owner,
          row.reason,
          row.seedReason,
          row.segment,
          row.abc,
          productStatusValue(row),
          productStatusNote(row),
        ].filter(Boolean).join(" ").toLowerCase();
        if (!haystack.includes(search)) return false;
      }
      if (ownerFilter !== "all" && (row.owner || "Без owner") !== ownerFilter) return false;
      if (statusFilter !== "all" && row.decisionStatus !== statusFilter) return false;
      if (riskOnly && riskScore(row) < 3) return false;
      return true;
    });
    rows.sort((a, b) => {
      if (sortMode === "turnover") return (Number(b.turnoverDisplay) || 0) - (Number(a.turnoverDisplay) || 0);
      if (sortMode === "margin") return (Number(a.marginTotalPct) || 0) - (Number(b.marginTotalPct) || 0);
      if (sortMode === "plan") return (Number(a.completionPctSelected) || 0) - (Number(b.completionPctSelected) || 0);
      return riskScore(b) - riskScore(a) || (Number(b.turnoverDisplay) || 0) - (Number(a.turnoverDisplay) || 0);
    });
    return rows;
  }
