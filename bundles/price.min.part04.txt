  async function handleSubmit(event) {
    event.preventDefault();
    const form = event.currentTarget;
    const key = form.dataset.priceForm;
    const row = getCurrentRowByKey(key);
    if (!row) return;
    const previous = mergeEntryLayers().get(key);
    const entry = upsertLocalEntry({
      id: row.decisionId,
      marketplace: row.marketplace,
      articleKey: row.articleKey,
      decisionStatus: form.elements.decisionStatus.value,
      targetFillPrice: form.elements.targetFillPrice.value,
      targetClientPrice: form.elements.targetClientPrice.value,
      allowedMarginManualPct: parseNum(form.elements.allowedMarginManualPct.value) == null ? null : parseNum(form.elements.allowedMarginManualPct.value) / 100,
      reason: form.elements.reason.value,
      owner: form.elements.owner.value,
      dueDate: form.elements.dueDate.value,
      needsTask: form.elements.needsTask.checked,
      escalation: form.elements.escalation.value,
      comment: form.elements.comment.value,
      updatedBy: state.team.member?.name || "Команда",
      updatedAt: new Date().toISOString(),
      createdAt: previous?.createdAt || row.createdAt || new Date().toISOString(),
    });
    const history = addHistoryRow(entry, previous);
    if (entry.needsTask || entry.decisionStatus === "review" || entry.escalation !== "none") ensurePriceTask(row, entry);
    if (state.team.client && !state.priceWorkbench.remoteDisabled) {
      try {
        await upsertRemoteEntries([entry]);
        try {
          await insertRemoteHistory([history]);
          history.synced = true;
          saveLocalState();
        } catch (historyError) {
          console.error(historyError);
        }
      } catch (error) {
        console.error(error);
      }
    }
    state.team.note = "Решение по цене сохранено";
    updateSyncBadge();
    renderPriceWorkbench();
    const fresh = getCurrentRowByKey(key);
    if (fresh) renderPriceModal(fresh);
  }

  function handleCreateTask(event) {
    const key = event.currentTarget.dataset.priceTask;
    const row = getCurrentRowByKey(key);
    if (!row) return;
    const entry = normalizeEntry({
      id: row.decisionId,
      marketplace: row.marketplace,
      articleKey: row.articleKey,
      decisionStatus: row.decisionStatus,
      targetFillPrice: row.targetFillPrice,
      targetClientPrice: row.targetClientPrice,
      allowedMarginManualPct: row.allowedMarginManualPct,
      reason: row.reason,
      owner: row.owner,
      dueDate: row.dueDate,
      needsTask: true,
      escalation: row.escalation,
      comment: row.comment,
      createdAt: row.createdAt,
      updatedAt: new Date().toISOString(),
    });
    upsertLocalEntry(entry);
    ensurePriceTask(row, entry);
    state.team.note = "Задача по цене создана / обновлена";
    saveLocalState();
    saveLocalStorage();
    updateSyncBadge();
    renderPriceWorkbench();
    const fresh = getCurrentRowByKey(key);
    if (fresh) renderPriceModal(fresh);
  }

  function renderTaskSummary(tasks) {
    if (!tasks.length) return "";
    const counters = tasks.reduce((acc, task) => {
      const status = String(task.status || "new");
      acc[status] = (acc[status] || 0) + 1;
      return acc;
    }, {});
    const chips = Object.entries(counters)
      .sort((left, right) => right[1] - left[1])
      .map(([status, count]) => {
        const meta = TASK_STATUS_META[status] || { label: status, kind: "" };
        return badge(`${meta.label}: ${fmt.int(count)}`, meta.kind || "");
      })
      .join("");
    return chips ? `<div class="price-data-note">${chips}</div>` : "";
  }

  function renderPeriodHistory(row) {
    const dailyPlan = row.planUnitsSelected && row.selectedDays ? row.planUnitsSelected / row.selectedDays : (row.planMonthUnits && row.planMonthDays ? row.planMonthUnits / row.planMonthDays : null);
    if (!(row.monthlyFiltered || []).length) return `<div class="price-empty">За выбранный период нет истории.</div>`;
    return `
      <div class="price-history-wrap">
        <table class="price-history-table">
          <thead><tr><th>Дата</th><th>Заказы</th><th>Доставлено</th><th>Обор.</th><th>Цена MP</th><th>Δ MP</th><th>Цена клиента</th><th>Δ клиент</th><th>СПП</th><th>Выручка</th><th>% плана / день</th><th>Источник</th></tr></thead>
          <tbody>${row.monthlyFiltered.map((item, index, items) => {
            const actual = parseNum(item.ordersUnits) ?? parseNum(item.deliveredUnits);
            const dayCompletion = actual != null && dailyPlan ? actual / dailyPlan : null;
            const prev = index > 0 ? items[index - 1] : null;
            const prevPrice = parseNum(prev?.price);
            const prevClient = parseNum(prev?.clientPrice);
            const price = parseNum(item.price);
            const clientPrice = parseNum(item.clientPrice);
            const priceDelta = prevPrice != null && price != null && prevPrice > 0 ? (price - prevPrice) / prevPrice : null;
            const clientDelta = prevClient != null && clientPrice != null && prevClient > 0 ? (clientPrice - prevClient) / prevClient : null;
            const historySource = actual != null || parseNum(item.revenue) != null ? "daily-факт" : "цена / оборач.";
            return `<tr><td>${escapeHtml(String(item.date).slice(0, 10))}</td><td>${item.ordersUnits == null ? "—" : fmt.int(item.ordersUnits)}</td><td>${item.deliveredUnits == null ? "—" : fmt.int(item.deliveredUnits)}</td><td>${item.turnoverDays == null ? "—" : `${fmt.num(item.turnoverDays, 1)} дн.`}</td><td>${moneyShort(item.price)}</td><td>${priceDelta == null ? "—" : pctShort(priceDelta)}</td><td>${moneyShort(item.clientPrice)}</td><td>${clientDelta == null ? "—" : pctShort(clientDelta)}</td><td>${pctShort(item.sppPct)}</td><td>${moneyShort(item.revenue)}</td><td>${dayCompletion == null ? "—" : pctShort(dayCompletion)}</td><td>${escapeHtml(historySource)}</td></tr>`;
          }).join("")}</tbody>
        </table>
      </div>
    `;
  }
