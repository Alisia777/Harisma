  function pricePriorityScore(row) {
    let score = riskScore(row) * 10;
    if (row.marginTotalPct != null && row.effectiveAllowedMarginPct != null && Number(row.marginTotalPct) < Number(row.effectiveAllowedMarginPct)) score += 8;
    if (row.turnoverDisplay != null && Number(row.turnoverDisplay) >= 60) score += 6;
    else if (row.turnoverDisplay != null && Number(row.turnoverDisplay) >= 40) score += 3;
    if (row.completionPctSelected != null && Number(row.completionPctSelected) < 0.8) score += 5;
    if (row.needsTask) score += 4;
    if (row.decisionStatus === "review") score += 3;
    if (row.decisionStatus === "approved") score += 2;
    if (row.historyMode === "daily") score += 1;
    return score;
  }

  function pricePrimaryAction(row) {
    if (row.marginTotalPct != null && row.effectiveAllowedMarginPct != null && Number(row.marginTotalPct) < Number(row.effectiveAllowedMarginPct)) {
      if (row.requiredPriceForMargin != null) {
        return `Маржа ниже допустимой. Сначала проверьте, можно ли удержать цену не ниже ${moneyShort(row.requiredPriceForMargin)} без ухода за жесткий минимум и рабочую зону.`;
      }
      return "Маржа ниже допустимой. Нужна проверка коридора цены и причин, почему позиция ушла в минус относительно 3-месячной средней.";
    }
    if (row.turnoverDisplay != null && Number(row.turnoverDisplay) >= 60) {
      return "Позиция зависает по оборачиваемости. Сверьте рабочую зону цены, остаток и план продаж, прежде чем согласовывать новое решение.";
    }
    if (row.completionPctSelected != null && Number(row.completionPctSelected) < 0.8) {
      return "План по артикулу не дотягивает. Нужна проверка: это проблема цены, наличия, контента или ограничение по данным факта.";
    }
    if (row.needsTask) {
      return "По позиции уже зафиксировано действие. Проверьте owner, срок и статус, чтобы решение не зависало без исполнения.";
    }
    return "Критического отклонения нет. Используйте карточку для сверки Smart-цели, истории цены и принятого решения.";
  }

  function priceActionBullets(row) {
    const items = [];
    if (row.marginTotalPct != null && row.effectiveAllowedMarginPct != null && Number(row.marginTotalPct) < Number(row.effectiveAllowedMarginPct)) {
      const gap = Number(row.effectiveAllowedMarginPct) - Number(row.marginTotalPct);
      items.push(`Маржа ниже допустимой на ${pctShort(gap)}.`);
    }
    if (row.turnoverDisplay != null && Number(row.turnoverDisplay) >= 60) {
      items.push(`Оборачиваемость ${fmt.num(row.turnoverDisplay, 1)} дн. Требуется ускорение оборота.`);
    } else if (row.turnoverDisplay != null && Number(row.turnoverDisplay) >= 40) {
      items.push(`Оборачиваемость ${fmt.num(row.turnoverDisplay, 1)} дн. уже в зоне внимания.`);
    }
    if (row.completionPctSelected != null && Number(row.completionPctSelected) < 0.8) {
      const actual = row.actualUnitsSelected == null ? "факт не опубликован" : `${fmt.int(row.actualUnitsSelected)} шт.`;
      const plan = row.planUnitsSelected == null ? "план не загружен" : `${fmt.int(row.planUnitsSelected)} шт.`;
      items.push(`План-факт ${pctShort(row.completionPctSelected)}: ${actual} при плане ${plan}.`);
    }
    if (row.targetFillPrice != null && row.hardMinPrice != null && Number(row.targetFillPrice) < Number(row.hardMinPrice)) {
      items.push(`Текущее решение ниже жесткого минимума ${moneyShort(row.hardMinPrice)}.`);
    }
    if (row.historyMode === "price_only") {
      items.push("В истории доступны цена и оборачиваемость, но дневной факт продаж ограничен.");
    }
    if (row.marketplace === "wb" && row.currentClientPriceDisplay != null && row.currentSppDisplay != null) {
      items.push("По WB перепроверьте, не пришла ли клиентская цена уже со скидкой СПП.");
    }
    if (!items.length) items.push("Позиция выглядит стабильной и может использоваться как контрольная.");
    return items.slice(0, 4);
  }

  function priceSignalMarkup(row) {
    const items = signalsForRow(row);
    if (!items.length) return badge("Стабильно", "ok");
    return items.map((item) => badge(item.label, item.kind)).join("");
  }

  function pricePriorityRows(rows, limit = 6) {
    return rows
      .slice()
      .sort((left, right) => {
        return pricePriorityScore(right) - pricePriorityScore(left)
          || riskScore(right) - riskScore(left)
          || (Number(right.turnoverDisplay) || 0) - (Number(left.turnoverDisplay) || 0)
          || (Number(left.completionPctSelected) || 0) - (Number(right.completionPctSelected) || 0);
      })
      .slice(0, limit);
  }
