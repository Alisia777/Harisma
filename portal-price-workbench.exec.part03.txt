  function loadLocalState() {
    try {
      const entries = JSON.parse(localStorage.getItem(LOCAL_ENTRIES_KEY) || "[]");
      state.priceWorkbench.localEntries = Array.isArray(entries) ? entries.map(normalizeEntry) : [];
    } catch {
      state.priceWorkbench.localEntries = [];
    }
    try {
      const history = JSON.parse(localStorage.getItem(LOCAL_HISTORY_KEY) || "[]");
      state.priceWorkbench.localHistory = Array.isArray(history) ? history.map(normalizeHistory) : [];
    } catch {
      state.priceWorkbench.localHistory = [];
    }
  }

  function saveLocalState() {
    localStorage.setItem(LOCAL_ENTRIES_KEY, JSON.stringify(state.priceWorkbench.localEntries || []));
    localStorage.setItem(LOCAL_HISTORY_KEY, JSON.stringify(state.priceWorkbench.localHistory || []));
  }

  async function queryRemoteEntries() {
    if (!state.team.client) return [];
    const response = await state.team.client.from(ENTRIES_TABLE).select("*").eq("brand", currentBrand()).limit(2000);
    if (response.error) throw response.error;
    return response.data || [];
  }

  async function queryRemoteHistory() {
    if (!state.team.client) return [];
    const response = await state.team.client.from(HISTORY_TABLE).select("*").eq("brand", currentBrand()).order("changed_at", { ascending: false }).limit(4000);
    if (response.error) throw response.error;
    return response.data || [];
  }
