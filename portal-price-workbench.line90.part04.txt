  async function queryRemoteEntries() {
    if (!state.team.client) return [];
    const response = await state.team.client.from(ENTRIES_TABLE).select("*").eq("brand", currentBrand()).limit(2000);
    if (response.error) throw response.error;
    return response.data || [];
  }

  async function queryRemoteHistory() {
    if (!state.team.client) return [];
    const response = await state.team.client.from(HISTORY_TABLE).select("*").eq("brand", currentBrand()).order("changed_at", { ascending: false }).limit(4000);
    if (response.error) throw response.error;
    return response.data || [];
  }
